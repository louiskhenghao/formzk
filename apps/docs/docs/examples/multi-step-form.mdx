**Package:** `@formzk/mui` (also pulls in `@formzk/core`)

A wizard split across multiple steps, sharing a single form instance. The trick is **render every step up-front and hide inactive ones with `display: none`** — this keeps each `Controller` mounted, so values entered on earlier steps survive forward/back navigation.

Validation is run per-step via `form.trigger(stepFields)` from `useFormContext`, so users can't advance until the current step is valid. The final step submits the entire form.

    'App.tsx': `import React, { useState } from 'react';
import { useFormContext, useFormState, useWatch } from 'react-hook-form';
import { Formzk, Switch, Select } from '@formzk/mui';
import { yupResolver } from '@hookform/resolvers/yup';
import * as yup from 'yup';
import OutlinedInput from '@mui/material/OutlinedInput';
import {
  Alert,
  AlertTitle,
  Box,
  Button,
  Stack,
  Step,
  StepLabel,
  Stepper,
  Typography,
} from '@mui/material';

const schema = yup.object({
  email: yup.string().required('Email is required').email('Invalid email'),
  password: yup.string().required('Password is required').min(8, 'At least 8 characters'),
  firstName: yup.string().required('First name is required'),
  lastName: yup.string().required('Last name is required'),
  newsletter: yup.boolean().required(),
});

const stepLabels = ['Account', 'Profile', 'Preferences'];
const stepFields = [
  ['email', 'password'],
  ['firstName', 'lastName', 'role'],
  ['newsletter'],
];

// Hide-instead-of-unmount keeps every Controller alive so values persist
// across step navigation.
const StepPanel = ({ active, children }) => (
  <Box sx={{ display: active ? 'block' : 'none' }}>{children}</Box>
);

// Filter form errors down to the current step. yupResolver always validates
// the whole schema, so without this we'd flash errors for fields that live
// on later steps the user hasn't reached yet.
const StepErrors = ({ step }) => {
  const { errors } = useFormState();
  const fields = stepFields[step] ?? [];
  const messages = fields
    .map((f) => errors[f]?.message)
    .filter(Boolean);
  if (messages.length === 0) return null;
  return (
    <Alert severity="error" sx={{ mt: 2 }}>
      <AlertTitle sx={{ fontWeight: 'bold' }}>Please fix these issues</AlertTitle>
      {messages.map((m, i) => <Typography key={i}>{m}</Typography>)}
    </Alert>
  );
};

const StepNavigator = ({ step, setStep, isLast }) => {
  const { trigger } = useFormContext();
  const next = async () => {
    const ok = await trigger(stepFields[step], { shouldFocus: true });
    if (ok) setStep((s) => s + 1);
  };
  return (
    <Stack direction="row" sx={{ mt: 3, justifyContent: 'space-between' }}>
      <Button
        type="button"
        variant="outlined"
        disabled={step === 0}
        onClick={() => setStep((s) => s - 1)}
      >
        Back
      </Button>
      {isLast ? (
        <Formzk.MUI.Submit text="Complete" />
      ) : (
        <Button type="button" variant="contained" onClick={next}>
          Next
        </Button>
      )}
    </Stack>
  );
};

const Review = () => {
  const values = useWatch();
  return (
    <Box
      component="pre"
      sx={{ p: 1.5, bgcolor: '#0d1117', color: '#f0f6fc', borderRadius: 1, fontSize: 12 }}
    >
      {JSON.stringify(values, null, 2)}
    </Box>
  );
};

export default function App() {
  const [step, setStep] = useState(0);
  const [done, setDone] = useState(false);
  const isLast = step === stepLabels.length - 1;

  return (
    <Formzk.Native.Provider
      config={[
        { name: 'TextField', component: OutlinedInput, props: { fullWidth: true, size: 'small' } },
        { name: 'Switch', component: Switch },
        { name: 'Select', component: Select }
      ]}
    >
      <Box sx={{ p: 2, maxWidth: 640 }}>
        <Stepper activeStep={step} sx={{ mb: 3 }}>
          {stepLabels.map((l) => (
            <Step key={l}><StepLabel>{l}</StepLabel></Step>
          ))}
        </Stepper>

        {done && <Alert severity="success" sx={{ mb: 2 }}>Onboarding complete!</Alert>}

        <Formzk.MUI.Form
          name="onboarding"
          options={{
            mode: 'onSubmit',
            reValidateMode: 'onChange',
            resolver: yupResolver(schema),
            defaultValues: {
              email: '', password: '',
              firstName: '', lastName: '', role: '',
              newsletter: true,
            },
          }}
          onSubmit={(values) => { console.log(values); setDone(true); }}
        >
          <StepPanel active={step === 0}>
            <Stack spacing={2}>
              <Formzk.MUI.Item name="email" label="Email" component="TextField" />
              <Formzk.MUI.Item
                name="password"
                label="Password"
                component="TextField"
                props={{ type: 'password' }}
              />
            </Stack>
          </StepPanel>

          <StepPanel active={step === 1}>
            <Stack spacing={2}>
              <Formzk.MUI.Item name="firstName" label="First name" component="TextField" />
              <Formzk.MUI.Item name="lastName" label="Last name" component="TextField" />
              <Formzk.MUI.Item
                name="role"
                label="Role"
                component="Select"
                props={{
                  options: [
                    { label: 'Engineer', value: 'engineer' },
                    { label: 'Designer', value: 'designer' },
                    { label: 'PM', value: 'pm' },
                  ],
                }}
              />
            </Stack>
          </StepPanel>

          <StepPanel active={step === 2}>
            <Stack spacing={2}>
              <Formzk.MUI.Item
                name="newsletter"
                component="Switch"
                valueKey="checked"
                props={{ label: 'Subscribe to product updates' }}
              />
              <Review />
            </Stack>
          </StepPanel>

          <StepErrors step={step} />

          <StepNavigator step={step} setStep={setStep} isLast={isLast} />
        </Formzk.MUI.Form>
      </Box>
    </Formzk.Native.Provider>
  );
}
`,
  }}
/>

### What to try

- Fill out step 1, advance to step 2, then click **Back** — the email/password values are still there because every `Controller` stays mounted (just visually hidden).
- Try clicking **Next** without filling required fields — `trigger()` short-circuits the navigation and surfaces errors via `Formzk.MUI.Errors`.
- Inspect the live JSON on the review step — it updates immediately as you change fields on any step.
